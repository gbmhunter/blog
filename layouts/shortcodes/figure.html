{{ if not ($.Page.Scratch.Get "figurecount") -}}{{- end -}} {{- $.Page.Scratch.Add "figurecount" 1 -}} {{- if not (.Get "width") -}} {{- errorf "Width not specified for image '%s' on page '%s'." (.Get "src") ($.Page.Permalink) -}} {{- end -}} {{- $width_without_px := index (findRE "^[0-9]+" (.Get "width")) 0 -}} {{/* If the src to the image starts with a /, we assume it's not a page resource, and therefore can't resample it. If it is a page resource, resample to match the specified width */}} {{- $link := "" -}} {{ if hasPrefix (.Get "src") "/" }} {{ $link = .Get "src" }} {{ else }} {{- $image := .Page.Resources.GetMatch (printf "%s" (.Get "src")) -}} {{- if not $image -}} {{- errorf "Image resource not found for image '%s' on page '%s'." (.Get "src") ($.Page.Permalink) -}} {{- end -}} {{- $thumb := 0 -}} {{- if eq (index (split (.Get "src") ".") 1) "gif" -}} {{- $thumb = $image -}} {{- else if eq (index (split (.Get "src") ".") 1) "svg" -}} {{- $thumb = $image -}} {{- else -}} {{- $thumb = $image.Resize (printf "%s%s CatmullRom" $width_without_px "x") -}} {{- $link = $thumb.Permalink -}} {{- end -}} {{- end -}}
